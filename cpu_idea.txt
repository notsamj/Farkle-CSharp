createOptions(rollCounts, index, options, selectedCounts){
    // End recursion
    if (index === rollCounts.length){ return; }
    numOfThisFace = rollCounts[index]
    
    for (int i = 0; i < numOfThisFace+1; i++){
        copyOfSelectedCounts = copy(selectedCounts);
        copyOfSelectedCounts[index] = i;
        options.push(copyOfSelectedCounts);

        // Recurse
        createOptions(rollCounts, index+1, options, copyOfSelectedCounts);
    }
}
